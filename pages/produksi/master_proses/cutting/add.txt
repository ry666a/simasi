<script>
    $(document).ready(function() {
        $('#exact_code').select2({
            theme: 'bootstrap-5',
            placeholder: '-- Cari Produk dari Masterplan --',
            allowClear: true,
            width: '100%'
        });

        $('#exact_code').on('change', function() {
            let exactCode = $(this).val();

            if (exactCode && exactCode !== "") {
                $.ajax({
                    url: "<?= $base_url ?>/pages/produksi/master_proses/cutting/get_components.php",
                    type: 'POST',
                    data: {
                        exact_code: exactCode
                    },
                    dataType: 'json',
                    beforeSend: function() {
                        $('#componentContainer').fadeIn();
                        $('#componentList').html(`<tr><td colspan="5" class="text-center py-5"><i class="fas fa-spinner fa-spin fa-2x text-primary mb-2"></i><div class="text-muted">Loading data detail...</div></td></tr>`);
                        $('#btnSubmit').prop('disabled', true);
                    },
                    success: function(data) {
                        let html = '';
                        if (data && data.length > 0) {
                            data.forEach(function(item) {
                                html += `
                            <tr>
                                <td class="ps-4">
                                    <div class="fw-bold text-dark">${item.nama_komponen}</div>
                                    <input type="hidden" name="id_komponen[]" value="${item.id_komponen}">
                                </td>
                                <td class="text-center">
                                    <span class="badge-dimensi">${item.dimensi_panjang} x ${item.dimensi_lebar}</span>
                                </td>
                                <td class="text-center">
                                    <div class="edging-wrapper">
                                        <div class="edging-box box-panjang" title="Sisi Panjang">
                                            <div class="edging-label">P</div>
                                            <div class="edging-value">${item.sisi_panjang}</div>
                                        </div>
                                        <div class="edging-box box-pendek" title="Sisi Pendek">
                                            <div class="edging-label">S</div>
                                            <div class="edging-value">${item.sisi_pendek}</div>
                                        </div>
                                    </div>
                                </td>
                                <td class="text-center fw-bold text-primary">${item.jml_komponen}</td>
                                <td class="pe-4">
                                    <div class="runtime-container">
                                        <input type="number" 
                                            step="0.01" 
                                            name="runtime[]" 
                                            class="form-control input-runtime-custom text-center" 
                                            placeholder="0.00" 
                                            required
                                            value="${item.runtime}">
                                        <span class="runtime-unit-label">Min</span>
                                    </div>
                                </td>
                            </tr>`;
                            });
                            $('#btnSubmit').prop('disabled', false);
                        } else {
                            html = `<tr><td colspan="5" class="text-center py-5 text-muted"><i class="fas fa-info-circle fa-2x mb-2 text-warning"></i><br>Komponen tidak ditemukan.</td></tr>`;
                            $('#btnSubmit').prop('disabled', true);
                        }
                        $('#componentList').html(html);
                    }
                });
            } else {
                $('#componentContainer').hide();
            }
        });

        $('#formAddCutting').on('submit', function(e) {
            e.preventDefault();
            const btn = $('#btnSubmit');
            $.ajax({
                url: "<?= $base_url ?>/pages/produksi/master_proses/cutting/save.php",
                type: 'POST',
                data: $(this).serialize(),
                dataType: 'json',
                beforeSend: function() {
                    btn.prop('disabled', true).html('<i class="fas fa-spinner fa-spin me-2"></i>Saving...');
                },
                success: function(res) {
                    if (res.status === 'success') {
                        // Panggil Toast Success
                        showToast('Data Berhasil Disimpan!', 'success');
                        setTimeout(() => {
                 btn.prop('disabled', false).html('<i class="fas fa-check-circle me-2"></i>Simpan Master Proses');
            }, 1500);
                        
                    } else {
                        // Panggil Toast Error
                        showToast('Gagal: ' + res.message, 'error');
                        btn.prop('disabled', false).html('<i class="fas fa-check-circle me-2"></i>Simpan Master Proses');
                    }
                },
                error: function() {
                    showToast('Terjadi kesalahan koneksi server!', 'error');
                    btn.prop('disabled', false).html('<i class="fas fa-check-circle me-2"></i>Simpan Master Proses');
                }
            });
        });
    });
</script>
